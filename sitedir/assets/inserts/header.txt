				function openSesame() {

					//is the menu already opened?
					// the if condition below answers this.if it's detected,
					// everything in "if" is executed...
					if ($(".navdesktop").hasClass("opened")) {
						$("#floatinghead").attr('src','assets/head/bleh.gif');
						$(".navdesktop").removeClass('opened');
						$("#navmsg").html("&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&uparrow; head = menu opener");
